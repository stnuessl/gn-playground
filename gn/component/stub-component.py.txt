import argparse
import os
import sys

import clang.cindex


class Visitor:
    def __init__(self, compilation_database, sources):
        dirname = os.path.dirname(compilation_database)

        self.compilation_database = (
            clang.cindex.CompilationDatabase.fromDirectory(dirname)
        )
        self.sources = sources
        self.index = clang.cindex.Index.create()
        self.extern_fn_decls = {}
        self.fn_defs = {}

    def _get_translation_unit(self, source):
        entries = self.compilation_database.getCompileCommands(source)
        if not entries or len(entries) == 0:
            message = f'missing compile command for source "{source}"'
            raise RuntimeError(message)

        args = list(entries[0].arguments)

        return self.index.parse(
            None,
            args=args,
        )

    def _visit_fn_decl(self, cursor):
        if cursor.storage_class == clang.cindex.StorageClass.STATIC:
            return

        if cursor.is_definition():
            self.fn_defs[cursor.spelling] = cursor
            return

        self.extern_fn_decls[cursor.spelling] = cursor


    def _visit(self, cursor):
        if cursor.kind == clang.cindex.CursorKind.FUNCTION_DECL:
            self._visit_fn_decl(cursor)

        for child in cursor.get_children():
            self._visit(child)

    def run(self):
        for source in self.sources:
            tu = self._get_translation_unit(source)
            self._visit(tu.cursor)

        import json

        print(json.dumps(list(self.extern_fn_decls.keys())))
        print(json.dumps(list(self.fn_defs.keys())))
        undefined = {
            k: v for k, v in self.extern_fn_decls.items()
            if k not in self.fn_defs
        }
        print(json.dumps(list(undefined.keys()), indent=4))


def main():
    parser = argparse.ArgumentParser()
    parser.add_argument('--compilation-database')
    parser.add_argument('sources', nargs='+')
    parser.add_argument('-o')

    args = parser.parse_args()

    print(args.sources)
    print(args.compilation_database)

    visitor = Visitor(args.compilation_database, args.sources)
    visitor.run()

    sys.exit(0)


if __name__ == '__main__':
    main()
